  - type: Script
    text: |-
      list colors = [
          <1, 0, 0>, // red
          <0, 1, 0>, // green
          <0, 0, 1>  // blue
      ];

      // index of the current color
      integer i;

      default
      {
          touch_end(integer total_number)
          {   
              // get the color at the current index
              vector color = llList2Vector(colors, i);
              
              llSetPrimitiveParams([
                  PRIM_POINT_LIGHT, TRUE, color, 1, 10, 0.6
              ]);
             
              // increment the index to cycle through colors
              i = (i + 1) % llGetListLength(colors);
          }
      }
